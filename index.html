实现基于链表的插入排序算法，即由无序链表利用插入算法生成有序链表

#include <iostream>
#include <ctime>
#include <cstdlib>
using namespace std;
template <typename T>
class node  //建立单链表类
{
public:
T nodeValue;
node<T>*next;
node():next(NULL){}
node(constT&item,node<T>*nextNode=NULL)
:nodeValue(item),next(nextNode){}
};
node<int>*insert(node<int>*first) //插入排序算法函数，基于原链表进行插入排序操作 
{
if (first == NULL)  return first;
node<int>*head = new node<int>();//新建一个head指针，指向链表头节点的前面，即作为另一个新设的头节点（但他没有实际的值，只是便于寻找真正的头节点） 
head->next = first;
node<int>*cur = first,*next,*p,*q;
while (cur->next) //当cur指针的下一个元素不为空即还没遍历完所有的元素时进入循环 
{
next = cur->next; //新建next指针指向cur的下一个节点，代表将要进行插入操作的元素 
p = head;   //新建p指针指向head，指向已完成排序部分的元素 
while (p->next != next && p->next->nodeValue < next->nodeValue)
p = p->next; //如果p->next没有追上next指针并且已排序的元素值小于next的值，p就不断向后移动 
if (p->next != next) //如果在已排序部分链表中找到了大于插入元素的元素 
{                               
q = next->next;  //进行插入操作，新建q指针记住next元素的下一个节点的地址 
next->next = p->next;
p->next = next;
cur->next = q;
}
else {      //若找不到比插入元素大的节点，说明已经是升序了，cur指针向后移动 
cur = next;
}
}
return head->next; //返回插入排序之后的头节点 
} 
int main()
{
node<int>*first=NULL;      
srand((int)time(NULL)); //利用srand()函数随机生成链表的值 
for(int i=0;i<15;i++)
{
first=new node<int>(rand(),first);
}
node<int>*n1=first;    //新建n1指针指向链表头节点，保证first始终指向头节点 
cout<<"排序前的链表为："<<endl;  //输出原来的链表 
while(n1!=NULL)
{
cout<<n1->nodeValue<<"   ";
n1=n1->next;
}
cout<<endl;
node<int>*nodeNow=insert(first);  //调用插入排序函数并且新建nodeNow指针指向返回的头节点 
cout<<endl<<"排序后的链表为："<<endl;
while(nodeNow!=NULL)   //输出排序后的链表 
{
cout<<nodeNow->nodeValue<<"   ";
nodeNow=nodeNow->next;
}
}
